React Flexgrid

Installation

npm install react-flexgrid

Usage

const {Grid, Row, Col} = require('react-flexgrid');

const App = React.createClass({
  render() {
    return (
      <Grid>
        <Row>
          <Col xs={6} md={3}>Hello, world!</Col>
        </Row>
      </Grid>
    );
  }
});
